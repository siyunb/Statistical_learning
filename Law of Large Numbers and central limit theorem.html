一、中心极限定理验证

定义验证中心极限定理函数

函数默认值为实验次数为10,30,50,100次，每次试验抽取样本100个，利用0-1的均匀分布，

central <- function(n = c(10, 30, 50, 100), d = "runif", a = 1, b = 0) {
    if (length(n) == 1) 
        par(mfrow = c(1, 1))
    if (length(n) == 2) 
        par(mfrow = c(1, 2))
    if (length(n) >= 3) 
        par(mfrow = c(2, 2))  #根据向量大小分割作图框
    if (d == "runif") {
        for (i in 1:length(n)) {
            x <- matrix(runif(n[i] * 100, b, a), n[i], 100)
            s <- apply(x, 1, sum)  #1是行，2是列
            m = (a + b)/2  #计算均匀分布均值
            sd = (b - a)^2/12  #计算均匀分布方差
            s <- (s - 100 * m)/(sqrt(n[i]) * sqrt(sd))
            hist(s, xlim = c(-5, 5), ylim = c(0, 0.5), probability = T, col = "lightblue", 
                main = "均匀分布验证中心极限定理")
            lines(density(s, bw = 1), col = "blue", lwd = 3)
            y <- seq(-5, 5, length.out = 100)
            lines(y, dnorm(y, 0, 1), col = "red", lwd = 3)  #增加一条正态密度曲线
            legend("topright", c("标准正态", "核密度估计"), lwd = 2, col = c("red", 
                "blue"), cex = 0.6)  #增加图例
        }
    }
    if (d == "rt") {
        if (a <= 2) 
            stop("自由度要大于2")
        for (i in 1:length(n)) {
            x <- matrix(rt(n[i] * 100, a, b), n[i], 100)
            s <- apply(x, 1, sum)  #1是行，2是列
            m = 0  #计算t分布均值
            sd = a/(a - 2)  #计算t分布方差
            s <- (s - 100 * m)/(sqrt(n[i]) * sqrt(sd))
            hist(s, xlim = c(-5, 5), ylim = c(0, 0.5), probability = T, col = "lightblue", 
                main = "t分布验证中心极限定理")
            lines(density(s, bw = 1), col = "blue", lwd = 3)
            y <- seq(-5, 5, length.out = 100)
            lines(y, dnorm(y, 0, 1), col = "red", lwd = 3)  #增加一条正态密度曲线
            legend("topright", c("标准正态", "核密度估计"), lwd = 2, col = c("red", 
                "blue"), cex = 0.6)  #增加图例
        }
    }
    if (d == "rf") {
        if (b <= 4) 
            stop("第二自由度要大于4")
        for (i in 1:length(n)) {
            x <- matrix(rf(n[i] * 100, a, b, 0), n[i], 100)
            s <- apply(x, 1, sum)  #1是行，2是列
            m = b/(b - 2)  #计算f分布均值
            sd = 2 * b^2 * (a + b - 2)/a * (b - 2)^2 * (b - 4)  #计算f分布方差
            s <- (s - 100 * m)/(sqrt(n[i]) * sqrt(sd))
            hist(s, xlim = c(-5, 5), ylim = c(0, 0.5), probability = T, col = "lightblue", 
                main = "F分布验证中心极限定理")
            lines(density(s, bw = 1), col = "blue", lwd = 3)
            y <- seq(-5, 5, length.out = 100)
            lines(y, dnorm(y, 0, 1), col = "red", lwd = 3)  #增加一条正态密度曲线
            legend("topright", c("标准正态", "核密度估计"), lwd = 2, col = c("red", 
                "blue"), cex = 0.6)  #增加图例
        }
    }
    if (d == "rexp") {
        for (i in 1:length(n)) {
            x <- matrix(rexp(n[i] * 100, a), n[i], 100)
            s <- apply(x, 1, sum)  #1是行，2是列
            m = 1/a  #计算指数分布均值
            sd = 1/a^2  #计算指数分布方差
            s <- (s - 100 * m)/(sqrt(n[i]) * sqrt(sd))
            hist(s, xlim = c(-5, 5), ylim = c(0, 0.5), probability = T, col = "lightblue", 
                main = "指数分布验证中心极限定理")
            lines(density(s, bw = 1), col = "blue", lwd = 3)
            y <- seq(-5, 5, length.out = 100)
            lines(y, dnorm(y, 0, 1), col = "red", lwd = 3)  #增加一条正态密度曲线
            legend("topright", c("标准正态", "核密度估计"), lwd = 2, col = c("red", 
                "blue"), cex = 0.6)  #增加图例
        }
    }
    if (d == "rbinom") {
        for (i in 1:length(n)) {
            x <- matrix(rbinom(n[i] * 100, 1, a), n[i], 100)
            s <- apply(x, 1, sum)  #1是行，2是列
            m = a  #计算二项分布均值
            sd = 100 * a * (1 - a)  #计算二项分布方差
            s <- (s - 100 * m)/(sqrt(n[i]) * sqrt(sd))
            hist(s, xlim = c(-5, 5), ylim = c(0, 0.5), probability = T, col = "lightblue", 
                main = "二项分布验证中心极限定理")
            lines(density(s, bw = 1), col = "blue", lwd = 3)
            y <- seq(-5, 5, length.out = 100)
            lines(y, dnorm(y, 0, 1), col = "red", lwd = 3)  #增加一条正态密度曲线
            legend("topright", c("标准正态", "核密度估计"), lwd = 2, col = c("red", 
                "blue"), cex = 0.6)  #增加图例
        }
    }
}

二、多功能绘图函数

定义函数

#data:prefer a data frame or a vector of characters or numbers.
#by:aggregate by this vector
#type:which type of plot you want to make.
#main:the mail title of your plot.
#xlab:same as in plot.
#ylab:same as in plot.
#prob:in histplot,if you want a histplot with probability,set prob = TRUE
#type you can input:scatterplot,barplot,histplot,boxplot,pie 
intelPlot<-function(data,by = NULL,type = NULL,aggreFun = NULL,main = NULL,xlab = NULL,ylab = NULL,prob = FALSE,addlines = FALSE)
{
  if(!is.null(data) & is.null(by))
  {
    #数值型
    if((class(data) == 'numeric') | (class(data) == 'integer'))
    {
      if(is.null(type))
        hist(data,main = main,xlab = xlab,ylab = ylab,col = 'lightgreen')
      ##hist plot
      else if(type == 'histplot')
      {
        if(!addlines)
        {
          hist(data,main = main,xlab = xlab,ylab = ylab,probability = prob,col = 'lightgreen')
        }
        if(addlines)
        {
          hist(data,main = main,xlab = xlab,ylab = ylab,probability = TRUE,col = 'lightgreen')
          lines(density(data))
        }
      }
      ##scatter plot
      else if(type == 'scatterplot')
        plot(data,main = main,xlab = xlab,ylab =ylab)
      #boxplot
      else if(type == 'boxplot')
        boxplot(data,main = main,xlab = xlab,ylab = ylab,col = 'lightgreen')
      #茎叶图
      else if(type == 'stem')
        stem(data)
    }
    
    #字符型或者因子型
    if((class(data) == 'character') | (class(data) == 'factor'))
    {
      if(is.null(type))
        barplot(table(data),main = main,xlab = xlab,ylab = ylab,col = rainbow(length(unique(data))))
      else if(type == 'barplot')
        barplot(table(data),main = main,xlab = xlab,ylab = ylab,col = rainbow(length(unique(data))))
      else if(type == 'pie')
        pie(table(data),main= main,xlab = xlab,ylab = ylab,col = rainbow(length(unique(data))))
    }
  }
  #此处传入两个向量
  if(!is.null(data) & (!is.null(by)))
  {
    if(class(data) %in% c('numeric','integer') & 
       class(by) %in% c('character','factor'))
    {
      #以by为列名，data为数据绘制柱状图,前提是by不重复
      if(length(unique(by)) == length(by))
        barplot(data,names.arg = by,main = main,xlab = xlab,ylab = ylab,col = rainbow(length(by)))
      #by有重复值，则根据by列对data进行聚合运算，绘制柱状图
      if(length(unique(by))!=length(by))
      {
        if(type == 'barplot' || is.null(type))
        {
          newD <- sapply(unique(by),function(x){
          aggreFun(data[by==x])
        })
        barplot(newD,names.arg = unique(by), main = main,xlab = xlab,ylab = ylab,col = rainbow(length(newD)))
        }
        else if(type =='boxplot')
        {
          newD <- data.frame(data,by)
          boxplot(data~by,newD,main = main,xlab = xlab,col = rainbow(length(unique(by))))
        }
      }
        
    }
    if(class(data) %in% c('numeric','integer') & 
       class(by) %in% c('numeric','integer'))
    {
      if(type == 'scatterplot')
        plot(data,by,main = main,xlab = xlab,ylab =ylab)
      else if(type == 'line')
        plot(data,by,type = 'l',main = main,xlab = xlab,ylab =ylab)
     
        
    }
    if(class(data) %in% c('numeric','integer') & 
      class(by) == 'Date')
    {
      plot(by,data,type = 'l',main = main,xlab = xlab,ylab = ylab,xlim = range(by),xaxt = 'n')
      axis(side = 1,at = by,labels = by)
    }
  }
}
